 * 
 * @details This file provides functions to toggle (flip) bits at specified[9D[K
specified positions in various integer types.
 * Toggle operation performs XOR with a bitmask to flip the bit from 0 to 1[1D[K

 * Supports 8-bit, 16-bit, 32-bit, and 64-bit signed and unsigned integers.[9D[K

#include <stdint.h>
#include <stdio.h>
#include <stdbool.h>

#ifndef METACORE___CLITE_BITUTILITYS_TOGGLE_BIT_H
#define METACORE___CLITE_BIT.utilityS-toggle_bit_h

/**
 * @brief Toggle (flip) bit at specified position in 8-bit number
 * 
 * @param num The value to modify
 * @param position Bit position (0-7)
 * @return Modified value with bit toggled
 */
uint8_t toggle_u8bit(uint8_t num, uint8_t position) {
    if (position > 7) {
        printf("Error: position must be 0-7\n");
        return num;
    }
    return num ^ (1 << position);
}

/**
 * @brief Toggle (flip) bit at specified position in 16-bit number
 * 
 * @param num The value to modify
 * @param position Bit position (0-15)
 * @return Modified value with bit toggled
 */

    if (position > 15) {
        printf("Error: position must be 0-15\n");
        return num;
    }
    return num ^ (1U << position);
}
